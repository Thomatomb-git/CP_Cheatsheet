For any positive integer $n$,
\[
\mu(n) = 
\begin{cases} 
1 & \text{if } n = 1, \\
0 & \text{if } a^2 \mid n \text{ for some } a > 1, \\
(-1)^k & \text{if } n \text{ is the product of } k \text{ distinct primes. } \square
\end{cases}
\]
\begin{lstlisting}
  int mobius[N + 1];
  void precompute_mobius(){
    for (int i = 1; i <= N; i++) {
      mobius[i] = 1;
    }
    for (int i = 2; i <= N; i++) {
      if (mobius[i] == 1) {
        for (int j = i; j <= N; j += i) {
          if (j % (i * i) == 0) {
            mobius[j] = 0;
          }
          else {
            mobius[j] *= -1;
          }
        }
      }
    }
  }
\end{lstlisting}
